import cv2

face = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')

#5.RATING PROVIDER:
import pandas as pd
from textblob import TextBlob
data = {
'feedback': [
"I love this product! It works great.",
"This is the worst experience I've ever had.",
]
}
df = pd.DataFrame(data)

def rate(feedback):
    analysis = TextBlob(feedback)

    if analysis.sentiment.polarity > 0.5:
        return 5 
    elif analysis.sentiment.polarity > 0:
        return 4 
    elif analysis.sentiment.polarity == 0:
        return 3 
    elif analysis.sentiment.polarity > -0.5:
        return 2 
    else:
        return 1 
    
df['rating'] = df['feedback'].apply(rate)
print(df)

#7.AIR QUALITY DETECTION :
import pandas as pd, numpy as np
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import mean_absolute_error, mean_squared_error

df = pd.read_csv(r"E:\SEM5\MAI\MAI Lab\MAI LAB Datasets\Airquality.csv", decimal=',')
df.replace(-200, np.nan, inplace=True)
df.fillna(df.mean(numeric_only=True), inplace=True)

X = StandardScaler().fit_transform(df[['PT08.S1(CO)','PT08.S2(NMHC)','PT08.S3(NOx)',
                                      'PT08.S4(NO2)','PT08.S5(O3)','T','RH','AH']])
Y = ['CO(GT)','NMHC(GT)','NOx(GT)','NO2(GT)','C6H6(GT)']

for y in Y:
    print("\nTraining Linear Regression for:", y)
    Xtr,Xts,Ytr,Yts = train_test_split(X, df[y], test_size=0.2, random_state=42)
    
    m = LinearRegression().fit(Xtr,Ytr)
    p = m.predict(Xts)
    print("MAE:", round(mean_absolute_error(Yts,p),2), "| MSE:", round(mean_squared_error(Yts,p),2))
